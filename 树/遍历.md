## 二叉树的先序、中序、后序、层次遍历
先序和中序都是“一路向左，走完走右”，不同点在于先序遍历是在入栈的时候访问，中序遍历是在出栈的时候访问
```cpp
struct node{
	int val;
	node* left = NULL;
	node* right = NULL;
};
void pre(node* t)
{
	stack<node*> s;
	while(t || !s.empty())
	{
		while(t) //一路向左 
		{
			cout<<t->val;
			s.push(t);
			t = t->left;
		}
		if(!s.empty()) //左边走完了走右 
		{
			node* temp = s.top();
			s.pop();
			t = t->right;
		}
	}
}
```
```cpp
void in(node* t)

}
```

层次遍历
```cpp
void level(node* t)
{
	queue<node*> q;
	q.push(t);
	while(!q.empty())
	{
		t = q.front();
		cout<<t->val;
		q.pop();
		if(t->left)  //左儿子入队 
		q.push(t->left);
		if(t->right)  //右儿子入队 
		q.push(t->right);
	}
}

//按层输出的层次遍历
void level(node* t)
{
	queue<node*> q;
	q.push(t);
	while(!q.empty())
	{
		int size = q.size();
		for(int i = 0;i < size;++i)
		{
			node* t = q.front();
			cout<<t->val<<" ";
			q.pop();
			if(t->left)
			q.push(t->left);
			if(t->right)
			q.push(t->right);
		}
		cout<<"# "; //层数++ 
	}
}
```
## 遍历序的互相转换
![](https://github.com/tangshisong/algorithm/blob/master/pic/4.png)  
两种方法：（1）先建树后遍历，此方法时间空间开销大 （2）根据序的特点进行划分  
根据先序和中序求后序
```cpp
int pre[] = {1,2,3,4,5,6,7,8,9};
int post[] = {3,2,5,8,7,9,6,4,1};
int in[] =   {2,3,1,5,4,7,8,6,9};
//这个代码很重要一定要掌握
void postorder(int preL,int preR,int inL,int inR)//求后序
{
	//递归边界 
	if(preL > preR)
	return;
	
	//找根 
	int t;
	for(t = inL;t <= inR;++t)
	if(in[t] == pre[preL])
	break;
	
	//子树划分 
	postorder(preL + 1,preL + t - inL ,inL,t-1); //t-inL为左子树的长度 
	postorder(preL + t - inL + 1,preR,t+1,inR);
	
	//访问 
	printf("%d ", pre[preL]);
}
```
根据后序和中序求先序
```cpp
void preorder(int postL,int postR,int inL,int inR)//求先序

	return ;
	//找根
	int t;

	if(post[postR] == in[t])
	break;
	//访问
	printf("%d ", post[postR]);
	//子树划分
	preorder(postL,postL + t - inL - 1,inL,t-1);
	preorder(postL + t - inL,postR-1,t+1,inR);
}
```
根据后序和中序求层次序
```cpp
#include <cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
//测试数据
int pre[] = {1,2,3,4,5,6,7,8,9};
int post[] = {3,2,5,8,7,9,6,4,1};
int in[] = {2,3,1,5,4,7,8,6,9};

struct node{
	int val;
	int idx;
	node(int x,int y)
	{
		val = x;
		idx = y;
	}
};
bool  cmp(node a,node b)
{
	return a.idx < b.idx;
}
vector<node> v;
int index = 1;
void level(int postL,int postR,int inL,int inR,int index) //后序和中序求层次 
{
	if(postL > postR)
	return ;
	int t;
	for(t = inL;t <= inR;++t) //这个循环可以用map进行优化 
	if(post[postR] == in[t])
	break;
	v.push_back(node(post[postR],index));
	level(postL,postL+t-inL-1,inL,t-1,2*index);
	level(postL+t-inL,postR-1,t+1,inR,2*index+1);													
}
int main()
{
	level(0,8,0,8,index);
	sort(v.begin(),v.end(),cmp);
	for(int i = 0;i < 9;++i)
	cout<<v[i].val<<" ";
}
```
## 根据遍历序建树
稍微修改上面的代码即可
```cpp
node* create(int preL,int preR,int inL,int inR)
{
	if(preL > preR)
	return NULL;
	int t;
	for(t = inL;t <= inR;++t)
	if(pre[preL] == in[t])
	break;
	node* root = new node(pre[preL]);
	root->left = create(preL+1,preL+t-inL,inL,t-1);
	root->right = create(preL+t-inL+1,preR,t+1,inR);
	return root;
}
```
## 树的遍历
N叉树的前序和后序与二叉树不同在于是用vector存储儿子结点
```cpp
struct node{
	int val;
	vector<node*> child;
};
void preorder(node* t)
{
	if(t)
	{
		cout<<t->val<<" ";
		for(int i = 0;i < t->child.size();++i)
		preorder(t->child[i]);
	}
}
```
