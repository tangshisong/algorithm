# [右边第一个大于的元素](https://leetcode-cn.com/problems/daily-temperatures/)
采用单调递减栈
```cpp
    vector<int> dailyTemperatures(vector<int>& T) {
        int n = T.size();
        vector<int> ans(n,0);
        stack<int> s;
        s.push(0);
        int i = 1;
        while(i < n)
        {
            while(!s.empty()&&T[s.top()] < T[i])
            {
                ans[s.top()] = i - s.top();
                s.pop();
            }
//栈为空或者是递减序列
            s.push(i);
            ++i;

        }

        return ans;
    }
```
