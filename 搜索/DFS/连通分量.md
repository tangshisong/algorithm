## [网格搜索](https://ac.nowcoder.com/acm/problem/14591)
题目大意：找出一个连通分量的权值总和为m的。  
思路：多次dfs就行了，代码完全按照模板写
```cpp
#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<string>
#include<stack>
#include<queue>
#define N 100
using namespace std;

int dir[4][2] = {0,1,0,-1,1,0,-1,0};
int map[N][N],vis[N][N];
int sum = 0;

void dfs(int x,int y)
{
	sum += map[x][y];
	
	for(int i = 0;i < 4;++i)
	{
		int xx = x + dir[i][0];
		int yy = y + dir[i][1];
		if(xx>=0&&yy>=0&&!vis[xx][yy]&&map[xx][yy])
		{
			vis[xx][yy] = 1;
			dfs(xx,yy);
		}
	}
}

int main()
{
	int n,m;
	while(cin>>n>>m&&n&&m)
	{
		for(int i = 0;i < n;++i)
		for(int j = 0;j < n;++j)
		{
			cin>>map[i][j];
			vis[i][j] = 0;
		}
		
		bool flag = false;
		for(int i = 0;i < n;++i)
		{
			if(flag)
			break;
			for(int j = 0;j < n;++j)
			if(map[i][j]&&!vis[i][j])
			{
				sum = 0;
				vis[i][j] = 1;
				dfs(i,j);
				if(sum == m)
				{
					cout<<"YES"<<endl;
					flag = true;
					break;
				}
			}
		}
		if(!flag)
		cout<<"NO"<<endl;
	}
}
```
