### 题目大意：给定正整数n，求长度为n的字符串s，其中s是由1、2、3组成的字符串，s不能存在连续3个相同的数字（比如1113），求所有s的个数。   

```cpp
#include<iostream>
#include<algorithm>
#include<vector>
#include<map>
#define N 1000010
using namespace std;
//dp[i][j][k]代表前i个字符末尾是j且j出现k次的个数 
static long long dp[N][4][3] = {0};
long long mod = 1e10;
int flag = 0;
int main()
{
	long long n;
	cin>>n;
	dp[1][1][1] = dp[1][2][1] = dp[1][3][1] = 1;
	
	for(int i = 2;i <= n;++i)
	for(int j = 1;j <= 3;++j)
	for(int k = 1;k <= 2;++k)
	{
		if(k == 1)
		for(int l = 1;l <= 3;++l)
		if(l != j)
		dp[i][j][k] += (dp[i-1][l][k]+dp[i-1][l][k+1]);
		
		if(k == 2)
		dp[i][j][k] += dp[i-1][j][k-1]; 
		
		if(dp[i][j][k] >= 1e16 || flag)
		{
			flag = 1;
			dp[i][j][k] %= mod;
		}
	}
	
	long long ans = 0;
	for(int i = 1;i <= 3;++i)
	for(int j = 1;j <= 2;++j)
	if(flag)
	ans = (ans + dp[n][i][j]) % mod;
	else
	ans += dp[n][i][j];
	
	if(!flag)
	cout<<ans;
	else
	{
		cout<<"......"<<ans;
	}
}
```

```cpp
#include<iostream>
#include<algorithm>
#include<vector>
#include<map>
using namespace std;
int n,temp=0;
map<int,int> mp; 

void dfs(int u,int father,vector<vector<int> >&tree,vector<int>& val,vector<int>& ans)
{
	if(mp[val[u]] == 0)
	++temp;
	++mp[val[u]];
	ans[u] = temp;
	
	for(int v = 1;v <= n;++v)
	if(tree[u][v] && v != father)//为什么有father：防止dfs的时候重复向上搜 
	{
		dfs(v,u,tree,val,ans);
	}
	
	--mp[val[u]];
	if(mp[val[u]] == 0)
	--temp;
}
int main()
{
	int a,b;
	cin>>n;
	vector<vector<int> > tree(n+1,vector<int>(n+1,0));
	vector<int> val(n+1);
	vector<int> ans(n+1);
	for(int i = 0;i < n-1;++i)//n-1条边 
	{
		cin>>a>>b;
		tree[a][b] = tree[b][a] = 1;
	}
	for(int i = 0;i < n;++i)//n个点的值 
	cin>>val[i+1];
	for(int i = 1;i <= n;++i)
	mp[val[i]] = 0;
	
	dfs(1,1,tree,val,ans);
	
	for(int i = 1;i < n;++i)
	cout<<ans[i]<<" ";
	cout<<ans[n];
}
```
