#include<iostream>
#include<algorithm>
#include<vector>
#include<map>
using namespace std;
int n,temp=0;
map<int,int> mp; 

void dfs(int u,int father,vector<vector<int> >&tree,vector<int>& val,vector<int>& ans)
{
	if(mp[val[u]] == 0)
	++temp;
	++mp[val[u]];
	ans[u] = temp;
	
	for(int v = 1;v <= n;++v)
	if(tree[u][v] && v != father)//为什么有father：防止dfs的时候重复向上搜 
	{
		dfs(v,u,tree,val,ans);
	}
	
	--mp[val[u]];
	if(mp[val[u]] == 0)
	--temp;
}
int main()
{
	int a,b;
	cin>>n;
	vector<vector<int> > tree(n+1,vector<int>(n+1,0));
	vector<int> val(n+1);
	vector<int> ans(n+1);
	for(int i = 0;i < n-1;++i)//n-1条边 
	{
		cin>>a>>b;
		tree[a][b] = tree[b][a] = 1;
	}
	for(int i = 0;i < n;++i)//n个点的值 
	cin>>val[i+1];
	for(int i = 1;i <= n;++i)
	mp[val[i]] = 0;
	
	dfs(1,1,tree,val,ans);
	
	for(int i = 1;i < n;++i)
	cout<<ans[i]<<" ";
	cout<<ans[n];
}
